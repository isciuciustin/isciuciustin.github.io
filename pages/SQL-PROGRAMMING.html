    Type of Triangle 
                                        
                                        SELECT 
                                        CASE 
                                            WHEN (A >= B + C) OR (B >= A + C) OR (C >= A + B) THEN 'Not A Triangle' 
                                            WHEN ((A = B) AND (A != C)) OR ((A = C) AND (A != B)) OR ((B = C )AND( A != B)) THEN 'Isosceles'
                                            WHEN A = B AND A = C AND B = C THEN 'Equilateral'
                                            ELSE 'Scalene'
                                        END 
                                        FROM TRIANGLES;
    The PADS
