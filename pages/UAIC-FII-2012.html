                                        #include < iostream >
                                        #include <cstring >
                                        using namespace std;
                                        
                                        char s[105], S[105][105], P[105][105], C[105][105];
                                        int szS, szP, szC;
                                        
                                        int main(){
                                            while(cin>>s){
                                                char cv[105];
                                                int it = 0, sz = 0;
                                                while(s[it] != '#'){
                                                    cv[sz++] = s[it++];
                                                }
                                                it++;
                                                cv[sz] = '\0';
                                                if(s[it] == 'S')
                                                    strcpy(S[szS++], cv);
                                                if(s[it] == 'P')
                                                    strcpy(P[szP++], cv);
                                                if(s[it] == 'C')
                                                    strcpy(C[szC++], cv);
                                            }
                                        
                                            for(int i = 0; i < szS; i++)
                                                for(int j = 0; j < szP; j++)
                                                    for(int k = 0; k < szC; k++)
                                                       cout<< S[i]<<" "<<P[j]<<" "<<C[k]<<','<<" ";
                                            return 0;
                                        }
