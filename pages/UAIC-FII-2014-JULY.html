                                                        n[e] = 0;

                                                            if(sol[x][y] == 0)

                                                            else
                                                                n[sol[x][y]] = 1;
                                                    }
                                        
                                                    for(int k = 1; k <= sz; k ++){
                                                        int x = p[k].lin, y = p[k].col;
