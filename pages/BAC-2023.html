                                        int S(int x){
                                            int s = 0;
                                            for(int i = 1; i <= x / 2; i ++)
                                                if(x % i == 0)
                                                    s += i;
                                            s += x;
                                            return s;
                                        }   
                                        int abundent(int n){
                                            float val = S(n)/ n;
                                            bool g = false;
                                            for(int k = 1; k < (n - 1); k ++){
                                                float x = S(k) / k;
                                                if(x >= val)
                                                    g = true;
                                            }
                                            if(g)
                                                return 0;
                                            return 1;
                                        }
