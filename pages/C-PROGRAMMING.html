    PRINT CELSIUS TO FAHRENHEIT: 
                                        
                                        #include <stdio.h>

                                        /* print Celsius-Farenheit table 
                                            for celsius - 0, 20, ..., 300 */
                                        
                                        int main(){
                                            printf("print Celsius-Farenheit table\n");
                                        
                                            float fahr, celsius;
                                            int lower, upper, step;
                                        
                                            lower = 0;
                                            upper = 300;
                                            step = 20;
                                        
                                            celsius = lower;
                                            while(celsius <= upper){
                                                fahr = (9 * celsius + 160 )/ 5.0;
                                                printf("%3.0f\t%6.2f\n", celsius, fahr);
                                                celsius = celsius + step;
                                            }    
                                                
                                        }

    A PROGRAM THAT COPIES ITS INPUT TO ITS OUTPUT AND REPLACES \t, \b and \\ 
                                        
                                        #include <stdio.h>

                                        /* 
                                            write a program to that copies its input to its output, 
                                            replacing each tab by \t,each backspace by \b, 
                                            and each backslash by \\.
                                        */

                                        int main() {
                                            int c;

                                            while ((c = getchar()) != EOF) {
                                                if (c == '\t') {
                                                    putchar('\\');
                                                    putchar('t');
                                                } else if (c == '\b') {
                                                    putchar('\\');
                                                    putchar('b');
                                                } else if (c == '\\') {
                                                    putchar('\\');
                                                    putchar('\\');
                                                } else {
                                                    putchar(c);
                                                }
                                            }

                                            return 0;
                                        }
    A PROGRAM THAT PRINTS ITS INPUT ONE WORD PER LINE 
                                        
                                        #include <stdio.h>

                                        #define IN 0
                                        #define OUT 1

                                        int main() {
                                            int c, state = OUT;

                                            while ((c = getchar()) != EOF) {
                                                if(c == ' ' || c == '\n' || c == '\t')
                                                    state = OUT;
                                                else
                                                    if(state == OUT){
                                                        printf("%c", '\n');
                                                        state = IN;
                                                        printf("%c", c);
                                                    }
                                                    else{
                                                        printf("%c", c);
                                                    }
                                            }

                                            return 0;
                                        }
    A PROGRAM THAT PRINTS A HISTOGRAM OF THE LENGTHS OF WORDS 
                                        
                                        #include <stdio.h>

                                        int main() {
                                            int c, lw = 0, arr[1005], sz = 0;
                                            while ((c = getchar()) != EOF) {
                                                if(c == ' '){
                                                    if(lw){
                                                        arr[++sz] = lw;
                                                        lw = 0;
                                                    }
                                                }
                                                else
                                                    lw++;
                                            }
                                            for(int i = 1; i <= sz; i++){
                                                for(int j = 0; j < arr[i]; j++)
                                                    printf("%c", '-');
                                                printf("%c", '\n');
                                            }
                                            return 0;
                                        }
    A PROGRAM TO PRINT A HISTOGRAM OF THE FREQUENCES OF DIFFERENT CHARACTERS 
                                        
                                        #include <stdio.h>

                                        int fv[1005];

                                        int main(){
                                            int c, low = -1, upper = -1;
                                            while((c = getchar()) != '\n'){
                                                fv[(int)(c)]++;
                                                if(low == -1 || low > (int)(c))
                                                    low = (int)(c);
                                                if(upper == -1 || upper < (int)(c))
                                                    upper = (int)(c);
                                            }
                                            for(int i = low; i <= upper; i++)
                                                if(fv[i] > 0){
                                                    printf("%c : ", (char)(i));
                                                    for(int j = 0; j < fv[i]; j++)
                                                        printf("%c", '-');
                                                    printf("%c", '\n');
                                                }
                                        }
    A PROGRAM THAT DETERMINES THE LONGEST LINE OF TEXT 
                                        
                                        #include <stdio.h>
                                        #define MAXLINE 10005

                                        int getLine(char s[MAXLINE]);
                                        void copy(char to[], char from[]);

                                        int main(){
                                            int len;
                                            int max;
                                            char line[MAXLINE];
                                            char longest[MAXLINE];

                                            max = 0;
                                            
                                            while((len = getLine(line)) > 0){
                                                if(len > max){
                                                    max = len;
                                                    copy(longest, line);
                                                }
                                            
                                            }
                                            if(max > 0){
                                                printf("Longest string : %s", longest);   
                                            }
                                            return 0;
                                        }

                                        void copy(char to[], char from[]){
                                            int i = 0;
                                            while((to[i] = from[i]) != '\0')
                                                i++;
                                        }

                                        int getLine(char s[MAXLINE]){
                                            int c, i;

                                            for(i = 0; i < MAXLINE - 1 && (c = getchar()) != EOF && c != '\n'; i++){
                                                s[i] = c;
                                            }
                                            if(c == 'Z')
                                                return 0;
                                            if(c == '\n')
                                                s[i] = c;

                                            s[++i] = '\0';
                                            return i;
                                        }
    USING EXTERN VARIABLES 
                                        
                                        #include <stdio.h>
                                        #define MAXLINE 10005

                                        int len;
                                        int max;
                                        char line[MAXLINE];
                                        char longest[MAXLINE];

                                        int getLine(void);
                                        void copy(void);

                                        int main(){
                                            extern char line[];
                                            extern char longest[];

                                            max = 0;
                                            
                                            while((len = getLine()) > 0){
                                                if(len > max){
                                                    max = len;
                                                    copy();
                                                }
                                            
                                            }
                                            if(max > 0){
                                                printf("Longest string : %s", longest);   
                                            }
                                            return 0;
                                        }

                                        void copy(void){
                                            extern char line[], longest[];
                                            int i = 0;
                                            while((longest[i] = line[i]) != '\0')
                                                i++;
                                        }

                                        int getLine(void){
                                            int c, i;
                                            extern char line[];
                                            for(i = 0; i < MAXLINE - 1 && (c = getchar()) != 'Z' && c != '\n'; i++){
                                                line[i] = c;
                                            }
                                            if(c == 'Z')
                                                return 0;
                                            if(c == '\n')
                                                line[i] = c;

                                            line[++i] = '\0';
                                            return i;
                                        }
