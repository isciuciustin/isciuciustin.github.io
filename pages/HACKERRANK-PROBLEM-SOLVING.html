    Diagonal Difference 

                                    
                                    def diagonalDifference(arr):
                                    dp = 0
                                    ds = 0
                                    n = len(arr) - 1
                                    for i in range(0, len(arr)):
                                            dp += arr[i][i]
                                            ds += arr[i][n - i]
                                    difference = abs(dp - ds)
                                    return difference
